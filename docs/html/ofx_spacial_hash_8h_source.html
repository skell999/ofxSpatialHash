147inline std::vector<int>& ofxSpatialHash<T>::getNearestBuckets(float x, float y, float radius)
148{
149 m_bucketIndexReturnBuffer.clear();
150
151 // Screen space. Bounding box corners
152 float topLeftX = x - radius;
153 float topLeftY = y - radius;
154 float bottomRightX = x + radius;
155 float bottomRightY = y + radius;
156
157 // Grid space. Clip to stay inside grid
158 float gridTopLeftX = clip(topLeftX / m_cellWidth,0.f, m_gridSize - 1);
159 float gridTopLeftY = clip(topLeftY / m_cellHeight, 0.f, m_gridSize - 1);
160 float gridBottomRightX = clip(bottomRightX / m_cellWidth, 0.f, m_gridSize - 1);
161 float gridBottomRightY = clip(bottomRightY / m_cellHeight, 0.f, m_gridSize - 1);
162
163 // Bounding box dimensions
164 float width = (std::floorf(gridBottomRightX)) - (std::floorf(gridTopLeftX)) + 1.f;
165 float height = (std::floorf(gridBottomRightY)) - (std::floorf(gridTopLeftY)) + 1.f;
166
167 // Iterate over bounding box
168 for (size_t gx = 0; gx < width; gx++)
169 {
170 for (size_t gy = 0; gy < height; gy++)
171 {
172 // Translate from 0,0 to actual grid coordinates
173 float gridX = gx + std::floorf(gridTopLeftX);
174 float gridY = gy + std::floorf(gridTopLeftY);
175 // Get bucket index
176 int index = (gridY * m_gridSize) + gridX;
177 m_bucketIndexReturnBuffer.push_back(index);
178 }
179 }
180 return m_bucketIndexReturnBuffer;
181}
